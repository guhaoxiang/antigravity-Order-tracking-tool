═══════════════════════════════════════════════════════════════
  TEST B: 車型過濾邏輯（單元測試）
═══════════════════════════════════════════════════════════════

測試對象: slot-monitor/core/order-tracker.js → findNewOrdersInPausedZones()

執行指令:
  node -e "
    const { findNewOrdersInPausedZones } = require('./slot-monitor/core/order-tracker');
    const moment = require('moment-timezone');
    const TZ = 'Asia/Taipei';
    let passed = 0, failed = 0;

    function assert(name, cond) {
      if (cond) { console.log('  PASS: ' + name); passed++; }
      else { console.log('  FAIL: ' + name); failed++; }
    }

    const CATEGORIES = [
      { key: 'STANDARD_dropoff', label: '五人座送機', vehicleType: 'STANDARD', tripType: 'dropoff' },
      { key: 'STANDARD_pickup',  label: '五人座接機', vehicleType: 'STANDARD', tripType: 'pickup' },
      { key: 'LARGE_dropoff',    label: '七人座送機', vehicleType: 'LARGE',    tripType: 'dropoff' },
      { key: 'LARGE_pickup',     label: '七人座接機', vehicleType: 'LARGE',    tripType: 'pickup' },
    ];

    const ps = moment.tz('2026-04-10 14:00', TZ).unix();
    const pe = moment.tz('2026-04-10 16:00', TZ).unix();
    const pauseMap = {
      STANDARD_dropoff: [{ start: ps, end: pe }],
      STANDARD_pickup:  [{ start: ps, end: pe }],
      LARGE_dropoff:    [{ start: ps, end: pe }],
      LARGE_pickup:     [{ start: ps, end: pe }],
    };
    const noPrev = { STANDARD_dropoff: [], STANDARD_pickup: [], LARGE_dropoff: [], LARGE_pickup: [] };

    // B1: STANDARD 訂單只觸發 STANDARD
    console.log('B1: STANDARD 訂單只觸發 STANDARD 類別');
    var r = findNewOrdersInPausedZones(
      [{ id: 1001, reservationTime: ps+1800, requiredVehicleType: 'STANDARD' }],
      pauseMap, noPrev, CATEGORIES);
    assert('STANDARD_dropoff 有 #1001', r.STANDARD_dropoff.length === 1);
    assert('STANDARD_pickup 有 #1001', r.STANDARD_pickup.length === 1);
    assert('LARGE_dropoff 無觸發', r.LARGE_dropoff.length === 0);
    assert('LARGE_pickup 無觸發', r.LARGE_pickup.length === 0);

    // B2: LARGE 訂單只觸發 LARGE
    console.log('B2: LARGE 訂單只觸發 LARGE 類別');
    r = findNewOrdersInPausedZones(
      [{ id: 2001, reservationTime: ps+1800, requiredVehicleType: 'LARGE' }],
      pauseMap, noPrev, CATEGORIES);
    assert('STANDARD_dropoff 無觸發', r.STANDARD_dropoff.length === 0);
    assert('STANDARD_pickup 無觸發', r.STANDARD_pickup.length === 0);
    assert('LARGE_dropoff 有 #2001', r.LARGE_dropoff.length === 1);
    assert('LARGE_pickup 有 #2001', r.LARGE_pickup.length === 1);

    // B3: null 車型觸發全部
    console.log('B3: null 車型觸發所有類別');
    r = findNewOrdersInPausedZones(
      [{ id: 3001, reservationTime: ps+1800, requiredVehicleType: null }],
      pauseMap, noPrev, CATEGORIES);
    assert('STANDARD_dropoff 有 #3001', r.STANDARD_dropoff.length === 1);
    assert('STANDARD_pickup 有 #3001', r.STANDARD_pickup.length === 1);
    assert('LARGE_dropoff 有 #3001', r.LARGE_dropoff.length === 1);
    assert('LARGE_pickup 有 #3001', r.LARGE_pickup.length === 1);

    // B4: 訂單不在暫停時段內
    console.log('B4: 訂單時間不在暫停時段，不觸發');
    r = findNewOrdersInPausedZones(
      [{ id: 4001, reservationTime: ps-3600, requiredVehicleType: 'STANDARD' }],
      pauseMap, noPrev, CATEGORIES);
    assert('全部類別無觸發', r.STANDARD_dropoff.length === 0 && r.LARGE_dropoff.length === 0);

    // B5: 已知訂單不重複通知
    console.log('B5: 已知訂單不重複觸發');
    r = findNewOrdersInPausedZones(
      [{ id: 5001, reservationTime: ps+1800, requiredVehicleType: 'STANDARD' }],
      pauseMap, { STANDARD_dropoff: ['5001'], STANDARD_pickup: ['5001'], LARGE_dropoff: [], LARGE_pickup: [] },
      CATEGORIES);
    assert('STANDARD_dropoff 無觸發（已知）', r.STANDARD_dropoff.length === 0);

    // B6: 混合訂單正確分流
    console.log('B6: 多筆混合訂單正確分流');
    r = findNewOrdersInPausedZones([
      { id: 6001, reservationTime: ps+600,  requiredVehicleType: 'STANDARD' },
      { id: 6002, reservationTime: ps+1200, requiredVehicleType: 'LARGE' },
      { id: 6003, reservationTime: ps+1800, requiredVehicleType: null },
    ], pauseMap, noPrev, CATEGORIES);
    assert('STANDARD_dropoff 有 2 筆 (#6001,#6003)', r.STANDARD_dropoff.length === 2);
    assert('LARGE_dropoff 有 2 筆 (#6002,#6003)', r.LARGE_dropoff.length === 2);
    assert('STANDARD 不含 #6002', !r.STANDARD_dropoff.find(o => o.id === 6002));
    assert('LARGE 不含 #6001', !r.LARGE_dropoff.find(o => o.id === 6001));

    console.log('');
    console.log('結果: ' + passed + ' passed, ' + failed + ' failed');
    process.exit(failed > 0 ? 1 : 0);
  "

預期: 20 passed, 0 failed
